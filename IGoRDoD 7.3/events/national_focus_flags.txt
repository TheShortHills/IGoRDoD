country_event = {

	id = 370032

	trigger = {
		invention = pop_shift_invention_2
		NOT = {
			has_country_modifier = assim_tier_1
		}
	}
 
	mean_time_to_happen = {
		days = 1
	}
	
	title = "Assimilation Bonus 1"
	desc = "We understand the mind better"
	picture = "troy"

	option = {
		name = "Better assimilation!"
		add_country_modifier = {
				name = assim_tier_1
				duration = -1
		}
	}
}
